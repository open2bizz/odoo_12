Implement the Framework

To benefit from the default data in your module

By adding a few lines of code you'll provide your users with an easy way to fill data.

value = fields.Html(
	string = 'Value'
)
default_value = fields.Many2one(
     comodel_name = 'default.data',
     string = 'Default Value',
     domain = "[('model', '=', 'my.model'), ('field', '=', 'value')]"
)

@api.onchange("default_value")
def onchange_default_handshake(self):
	if self.default_handshake:
		update_data = self.default_handshake.get_update_default_data(self.handshake)
		if update_data['update']:
			self.value = update_data['data']
